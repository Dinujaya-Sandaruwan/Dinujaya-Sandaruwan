# Nee to know more details about me. 😋

<br/>

> Simple code to use my bio data API ( http://api.dinujaya.com/ ) <strong>hosted in Microsoft Azure Cloud VPS</strong>

``` tsx
import React from 'react';
import { QueryClient, QueryClientProvider, useQuery } from 'react-query';
import axios from 'axios';
import { Button, Card, Image, Text, VStack, HStack, Spinner } from '@chakra-ui/react';

interface BasicInfo {
  full_name: string;
  uer_name: string;
  nic_name: string;
  user_avater: string;
  age: number;
  country: string;
}

interface ContactDetails {
  address: string;
  postal_code: number;
  calls: string;
  whatsapp: string;
}

interface ApiResponse {
  basic_info: BasicInfo;
  contact_details: ContactDetails;
  current_ststus: string;
  skills: string[];
}

interface Props {
  fetchData: () => Promise<ApiResponse>;
}

const UserData: React.FC<Props> = ({ fetchData }) => {
  const { data, isLoading, error } = useQuery<ApiResponse>('userData', fetchData);

  if (isLoading) {
    return <Spinner />;
  }

  if (error) {
    return <div>Error fetching data</div>;
  }

  return (
    <div>
      <Button onClick={fetchData}>Get Data</Button>

      {data && (
        <Card>
          <VStack align="center">
            <Image src={data.basic_info.user_avater} boxSize="200px" borderRadius="full" />
            <Text fontWeight="bold">{data.basic_info.uer_name}</Text>
            <Text>{data.contact_details.calls}</Text>
            <Text>{data.contact_details.whatsapp}</Text>
            <Text>{data.current_ststus}</Text>
            <Text fontWeight="bold">Skills:</Text>
            <HStack>
              {data.skills.map((skill, index) => (
                <Text key={index}>{skill}</Text>
              ))}
            </HStack>
          </VStack>
        </Card>
      )}
    </div>
  );
};

const UserDataContainer: React.FC = () => {
  const fetchData = async () => {
    const response = await axios.get<ApiResponse>('http://api.dinujaya.com/');
    return response.data;
  };

  const queryClient = new QueryClient();

  return (
    <QueryClientProvider client={queryClient}>
      <UserData fetchData={fetchData} />
    </QueryClientProvider>
  );
};

export default UserDataContainer;


```
